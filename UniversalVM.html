Example
const uvm = require('uvm');

uvm.spawn({
    bootCode: `
        bridge.on('loopback', function (data) {
            bridge.dispatch('loopback', 'pong');
        });
    `
}, (err, context) => {
    context.on('loopback', function (data) {
        console.log(data); // pong
    });

    context.dispatch('loopback', 'ping');
});
